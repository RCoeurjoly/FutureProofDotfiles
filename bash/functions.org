git_sync_repo () {
    local repo_to_sync="$1"
    if [ -z "$repo_to_sync" ]; then
        echo "usage: git_sync_repo <repo_name>"
        return 1
    fi

    cd "$HOME/$repo_to_sync" || return 1
    git config --bool branch.master.sync true
    git config --bool branch.master.syncNewFiles true
    git-sync
}

git_sync_exocortex () {
    git_sync_repo "Exocortex"
}

git_sync_all () {
    local repos_to_sync="Exocortex LLPSI Metatron"
    local repo_to_sync

    for repo_to_sync in $repos_to_sync; do
        git_sync_repo "$repo_to_sync" || return 1
    done
}
